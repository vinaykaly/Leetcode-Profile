def findLeastNumOfUniqueInts(arr,k):

      dic = {}

      for i in arr:
            if i not in dic:
                  dic[i] = 1

            else:
                  dic[i] += 1

      values = sorted(dic.values())

      x = 0  # to track the index of values
      res = 0

      for i in values:
            if k>= i:
                  k = k-i
                  values[x] = 0

            else:
                  values[x] = k-i
                  k = 0

            if values[x] != 0:
                  res = res+1

            x = x+1
      return res
findLeastNumOfUniqueInts(arr = [4,3,1,1,3,3,2], k = 3)
